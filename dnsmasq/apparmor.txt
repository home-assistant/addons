#include <tunables/global>

profile dnsmasq flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  /usr/lib/bashio/bashio ix,

  # S6-Overlay
  /bin/** ix,
  /usr/bin/** ix,
  /usr/lib/bashio/** ix,
  /etc/s6/** ix,
  /run/s6/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix

  # Data access
  /data/** rw,

  /usr/sbin/dnsmasq cx,
}

profile /usr/bin/socat flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>
  #include <abstractions/nameservice>

  capability net_bind_service,
  capability setgid,
  capability setuid,
  capability dac_override,
  network inet raw,
  network inet6 raw,

  /etc/dnsmasq.conf rw,
  /run/*dnsmasq*.pid w,
  /run/dnsmasq/ r,
  /run/dnsmasq/* rw,
  /tmp/* rw,
}
